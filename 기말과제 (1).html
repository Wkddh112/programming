tag = Hannanum()
nouns = tag.nouns(data)
nouns = [n for n in nouns if len(n) > 1]

count = Counter(nouns)
tags = count.most_common(150)

icon = Image.open('picture.png').convert("RGBA")
mask = Image.new("RGB", icon.size, (255,255,255))
mask.paste(icon,icon)
mask = np.array(mask)

my_path='C:\Windows\\Fonts\\malgunbd.ttf'

wordcloud = WordCloud(font_path=my_path, 
                      background_color='white', 
                      width=1200, 
                      height=800,
                     mask=mask).generate_from_frequencies(dict(tags))

fig = plt.figure(figsize=(15,15))
plt.imshow(wordcloud, interpolation='bilinear')
plt.axis('off')
